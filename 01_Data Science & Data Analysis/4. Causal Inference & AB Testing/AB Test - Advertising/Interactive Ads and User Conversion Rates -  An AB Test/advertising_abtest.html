import scipy.stats as stats

delta = 0.01        # 
alpha = 0.05       # 
power = 0.8        # 
std_dev = 0.04      # 

beta = 1 - power
print(f"beta : {beta}")

#  z-  z-beta
z_alpha = stats.norm.ppf(1 - alpha/2) #
z_beta = stats.norm.ppf(1 - beta)

print(f"z-alpha : {z_alpha}")
print(f"z-beta : {z_beta}")

#

sample_2_group = n*2
